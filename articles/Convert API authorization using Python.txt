Convert API authorization using Python
<p>In order to use Convert's API, developers would first need to connect and authorize themselves. The code below is an example of how this can be done using Python.</p>
<pre>from datetime import datetime<br>import hmac<br>import hashlib<br>import base64<br>import requests<br>import binascii<br><br>ApplicationID = 'AAAAAAA'; #Replace it with your APP ID<br>ApplicationSecretKey = 'SSSSSSS'; #Replace it with your APP SECRET KEY<br><br>Expires = 30; # In seconds. Request lifetime. If your requests take much time, try to increase this value<br><br>RequestURL = "https://app.convert.com/api/v1/projects/";<br>RequestBody = '';<br><br>ExpiresTimestamp = int(datetime.now(tz=None).timestamp()) + Expires;<br><br>SignString = ApplicationID + "\n" + str(ExpiresTimestamp) + "\n" +RequestURL + "\n" +RequestBody;<br><br>message = bytes(SignString, 'utf-8')<br>secret = bytes(ApplicationSecretKey, 'utf-8')<br><br>signature = str(binascii.hexlify(hmac.new(secret, message, digestmod=hashlib.sha256).digest()),'ascii')<br>Authorization = "Convert-HMAC-SHA256 Signature=" + signature;<br><br>headers={"Expires": ExpiresTimestamp, "Convert-Authorization": Authorization, "Convert-Application-ID": ApplicationID}<br>r = requests.get(RequestURL,headers=headers)<br><br>print(r.json())</pre>
